from math import gcd
lista = []
listaskr=[]
for i in range(10,100):
  for j in range(1,19):
    if i//10 + i%10 == j:
      lista.append(f"{j}/{i}")
      listaskr.append(f"{int(j/(gcd(i,j)))}/{int(i/(gcd(i,j)))}")
print(f"lista szwajcarskich:{lista}")
print(f"lista skr√≥conych szwajcarskich:{listaskr}")
